PROGRAM hospital_admission

const dentistry = "dentistry"
const generalDoctor = "generalDoctor"
var scheduleList = []
const day = [1, 2, 3, 4, 5, 6, 7] // 1 mereprenstasikan senin dan 7 merepresentasikan minggu

FUNCTION InputPatientName (firstName, lastName)
    STORE "name" with firstName lastName
    RETURN "name"
END FUNCTION

FUNCTION InputPatientProfile (age, gender, phoneNumber)
    STORE "profile" with age, gender, phoneNumber
    RETURN "profile"
END FUNCTION

FUNCTION DoctorSchedule (day)
    STORE "schedule" with day
    RETURN "schedule"
END FUNCTION

    STORE "doctor" with dentistry

    STORE "patientProfile" with InputPatientProfile(49, male, +123456)
    STORE "doctorSchedule" with DoctorSchedule(Monday)
    IF "isPatientAvailable"
    STORE scheduleList with doctorSchedule

    WHILE "isPatientAvailable" = FALSE
        DO
        FOR "i" from 1 in "day"

                i += 1

ELSE
